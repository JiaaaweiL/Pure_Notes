### DFS
```cpp
class Solution {
public:
    
    int result = 0;
    int row_size, col_size;

    void DFS(vector<vector<char>>& grid, vector<vector<char>>& visited, int row, int col) {
        // 越界或已访问或当前是水
        if (row < 0 || row >= row_size || col < 0 || col >= col_size ||
            visited[row][col] == '1' || grid[row][col] == '0') {
            return;
        }

        // 标记为已访问
        visited[row][col] = '1';

        // 递归访问相邻的陆地
        DFS(grid, visited, row + 1, col);
        DFS(grid, visited, row - 1, col);
        DFS(grid, visited, row, col + 1);
        DFS(grid, visited, row, col - 1);
    }

    int numIslands(vector<vector<char>>& grid) {
        row_size = grid.size();
        col_size = grid[0].size();
        vector<vector<char>> visited(row_size, vector<char>(col_size, '0')); // 初始化 visited

        for (int i = 0; i < row_size; i++) {
            for (int j = 0; j < col_size; j++) {
                // 遇到新的岛屿
                if (grid[i][j] == '1' && visited[i][j] == '0') {
                    result++;                 // 岛屿数量增加
                    DFS(grid, visited, i, j); // 标记整座岛屿
                }
            }
        }
        return result;
    }
};
```

### BFS
```cpp
class Solution {
public:
    int result = 0;
    int row_size, col_size;

    void BFS(vector<vector<char>>& grid, vector<vector<char>>& visited, int row, int col) {
        queue<pair<int, int>> q;
        q.push({row, col});
        visited[row][col] = '1'; // 标记为已访问

        // 四个方向
        int directions[4][2] = {{1, 0}, {-1, 0}, {0, 1}, {0, -1}};

        while (!q.empty()) {
            auto [cur_row, cur_col] = q.front();
            q.pop();

            for (auto dir : directions) {
                int new_row = cur_row + dir[0];
                int new_col = cur_col + dir[1];

                // 检查是否越界、已访问、是水
                if (new_row >= 0 && new_row < row_size &&
                    new_col >= 0 && new_col < col_size &&
                    visited[new_row][new_col] == '0' && grid[new_row][new_col] == '1') {
                    q.push({new_row, new_col});
                    visited[new_row][new_col] = '1'; // 标记为已访问
                }
            }
        }
    }

    int numIslands(vector<vector<char>>& grid) {
        row_size = grid.size();
        col_size = grid[0].size();
        vector<vector<char>> visited(row_size, vector<char>(col_size, '0')); // 初始化 visited

        for (int i = 0; i < row_size; i++) {
            for (int j = 0; j < col_size; j++) {
                if (grid[i][j] == '1' && visited[i][j] == '0') {
                    result++;                // 新岛屿数量增加
                    BFS(grid, visited, i, j); // 用 BFS 遍历整座岛
                }
            }
        }
        return result;
    }
};
```
