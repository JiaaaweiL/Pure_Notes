### DFS

```cpp
class Solution {
public:
    bool dfs(int course, vector<vector<int>>& graph, vector<int>& visited) {
        if (visited[course] == 1) {
            return false; // 发现环
        }
        if (visited[course] == 2) {
            return true; // 已访问，无需重复处理
        }

        // 标记为正在访问
        visited[course] = 1;

        // 遍历当前课程的后续课程
        for (int i = 0; i < graph[course].size(); i++) {
            int nextCourse = graph[course][i];
            if (!dfs(nextCourse, graph, visited)) {
                return false; // 如果发现环，返回 false
            }
        }

        // 标记为已访问
        visited[course] = 2;
        return true;
    }
    bool canFinish(int numCourses, vector<vector<int>>& prerequisites) {
        vector<vector<int>> graph(numCourses);
        for (int i = 0; i < prerequisites.size(); i++) {
            int course = prerequisites[i][0];
            int prerequisite = prerequisites[i][1];
            graph[prerequisite].push_back(course);
        }

        // 2. 初始化访问状态数组
        vector<int> visited(numCourses, 0);

        // 3. 检测每个课程是否有环
        for (int i = 0; i < numCourses; i++) {
            if (!dfs(i, graph, visited)) {
                return false; // 如果发现环，返回 false
            }
        }
        return true; // 无环，返回 true
    }
};
```

### BFS
```cpp
class Solution {
public:
    bool canFinish(int numCourses, vector<vector<int>>& prerequisites) {
        // 1. 构建邻接表和入度表
        vector<vector<int>> graph(numCourses);
        vector<int> inDegree(numCourses, 0);

        for (int i = 0; i < prerequisites.size(); i++) {
            int course = prerequisites[i][0];
            int prerequisite = prerequisites[i][1];
            graph[prerequisite].push_back(course);
            inDegree[course]++;
        }

        // 2. 初始化队列，将入度为 0 的节点加入队列
        queue<int> q;
        for (int i = 0; i < numCourses; i++) {
            if (inDegree[i] == 0) {
                q.push(i);
            }
        }

        // 3. 进行拓扑排序
        int visitedCourses = 0; // 记录访问过的课程数量
        while (!q.empty()) {
            int course = q.front();
            q.pop();
            visitedCourses++;

            // 遍历当前课程的后续课程
            for (int i = 0; i < graph[course].size(); i++) {
                int nextCourse = graph[course][i];
                inDegree[nextCourse]--; // 入度减 1
                if (inDegree[nextCourse] == 0) {
                    q.push(nextCourse); // 如果入度为 0，则加入队列
                }
            }
        }

        // 4. 判断是否访问了所有课程
        return visitedCourses == numCourses;
    }
};
```
